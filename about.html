pad

Usage

  • Write JS code in the left pane to see the result (automatic refresh) in the right pane.
  • CTRL-S to save the script. Unique identifier added to the URL which can then be bookmarked.
  • Import other libraries by using the import statement (ESM compatible libraries)

Examples

console.log


import "https://unpkg.com/console-log-div";
import dayjs from "https://unpkg.com/dayjs@1.8.15/esm/index.js";

console.log(dayjs().format("DD-MM-YYYY"));

react


import {React, ReactDOM} from 'https://unpkg.com/es-react';
import nojsx from 'https://giabos.github.io/nojsx/index.js';
const {h1, div, p} = nojsx(React.createElement);


const app = div([h1('hello'), p("blabla")]);

ReactDOM.render(app,document.getElementById('root'));

superfine


import {h, patch} from 'https://unpkg.com/superfine';
import nojsx from 'https://giabos.github.io/nojsx/index.js';
const {h1, div, p} = nojsx(h);

patch(document.getElementById('root'), div([h1('hello'), p("blabla")]));
